CREATE PROCEDURE [dbo].[sp_GetEntrenamientoById]
    @idEntrenamiento NVARCHAR(25)
AS
BEGIN
    SELECT 
        *
    FROM 
        TEntrenamiento
    WHERE 
        idEntrenamiento = @idEntrenamiento;
END
