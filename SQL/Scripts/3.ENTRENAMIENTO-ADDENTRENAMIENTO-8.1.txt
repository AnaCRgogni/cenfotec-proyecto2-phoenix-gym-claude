ALTER PROCEDURE [dbo].[sp_addEntrenamiento]
	@Nombre VARCHAR(255),
    @IDCliente VARCHAR(50),
    @ReturnValue INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    -- Insert the new record
    INSERT INTO TEntrenamiento (Nombre, Fecha, IDCliente)
    VALUES (@Nombre, GETDATE(),  @IDCliente);

    -- Set the output parameter to the IDRutina of the newly inserted record
    SET @ReturnValue = SCOPE_IDENTITY();
END;
