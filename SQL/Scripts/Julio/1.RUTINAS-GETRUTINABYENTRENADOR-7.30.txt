Create PROCEDURE [dbo].[sp_getRutinasByEntrenador]
    @IDUsuario varchar (50)
AS
BEGIN
    SET NOCOUNT ON;

    SELECT IDRutina,Nombre,Fecha
	from TRutina
	where IDUsuario = @IDUsuario
END;
