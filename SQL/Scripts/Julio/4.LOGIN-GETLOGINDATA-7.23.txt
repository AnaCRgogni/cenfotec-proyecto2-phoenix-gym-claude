ALTER PROCEDURE [dbo].[sp_getLogInData]
    @emailUsuario nvarchar(25)
AS
BEGIN
    SET NOCOUNT ON;

    -- Obtén la cédula usando el correo electrónico
    DECLARE @cedula nvarchar(25);
    
    SELECT @cedula = [cedula]
    FROM [dbo].[TUsuario]
    WHERE [email] = @emailUsuario;
    
    -- Asegúrate de que se encontró una cédula
    IF @cedula IS NOT NULL
    BEGIN
        -- Obtén la información de la sal usando la cédula
        SELECT s.[salt], u.[cedula], u.[email], u.[rolUsuario], u.[contrasena],u.[estado]
        FROM [dbo].[TSalt] s
        INNER JOIN [dbo].[TUsuario] u ON s.[cedulaUsuario] = u.[cedula]
        WHERE s.[cedulaUsuario] = @cedula;
    END
    ELSE
    BEGIN
        -- Si no se encuentra la cédula, puedes devolver un mensaje de error o resultados vacíos
        -- Puedes usar un mensaje de error o un resultado vacío según sea necesario
        SELECT 'No se encontró el usuario con el correo proporcionado' AS ErrorMessage;
    END
END
