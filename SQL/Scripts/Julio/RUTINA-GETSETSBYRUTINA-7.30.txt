ALTER PROCEDURE [dbo].[sp_getSetsByRutina]
    @IDRutina INT
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        s.IDRutina,
        s.Sets,
        s.Peso,
        s.Tiempo,
        e.Nombre AS Nombre,
        e.Descripcion AS Descripcion,
        e.Imagen AS Imagen,
        m.Name AS Equipo
    FROM
        TSets s
    INNER JOIN
        TEjercicio e ON s.IDEjercicios = e.EjercicioID
    LEFT JOIN
        TMaquina m ON e.MaquinaID = m.Id
    WHERE
        s.IDRutina = @IDRutina;
END;
