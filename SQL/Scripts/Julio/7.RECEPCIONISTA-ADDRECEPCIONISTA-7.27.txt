-- Switch to the correct database
USE PhoenixFitness;
GO

--Check if the procedure already exist and frop it if it does
IF OBJECT_ID('dbo.sp_addRecepcionista','P') IS NOT NULL
BEGIN 
	DROP PROCEDURE dbo.sp_addRecepcionista;
END;
GO

-- Create the new or updated stored procedure 
CREATE PROCEDURE dbo.sp_addRecepcionista
	@cedula NVARCHAR(25),
	@rolUsuario NVARCHAR(25),
    @nombre NVARCHAR(25),
    @apellido1 NVARCHAR(25),
    @apellido2 NVARCHAR(25),
    @email NVARCHAR(50),
    @contrasena NVARCHAR(100),
    @telefono NVARCHAR(25),
    @fechaNacimiento DATE,
    @fotoIdCliente NVARCHAR(200),
    @fotoPerfilCliente NVARCHAR(200),
    @generoCliente NVARCHAR(25),
    @idMembresia INT,
	@estado bit,
    @ReturnValue INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    -- Check if email or cedula already exists
    IF EXISTS (SELECT 1 FROM TUsuario WHERE email = @email)
    BEGIN
        SET @ReturnValue = -1; -- Email already exists, set return value to -1
        RETURN; -- Exit the procedure
    END

    IF EXISTS (SELECT 1 FROM TUsuario WHERE cedula = @cedula)
    BEGIN
        SET @ReturnValue = -2; -- Cedula already exists, set return value to -2
        RETURN; -- Exit the procedure
    END

    -- Insert the new client
    INSERT INTO TUsuario (cedula, rolUsuario, nombre, apellido1, apellido2, email, contrasena, telefono, fechaNacimiento, fotoIdCliente, fotoPerfilCliente, generoCliente, idMembresia, estado)
    VALUES (@cedula, 'Recepcionista', @nombre, @apellido1, @apellido2, @email, @contrasena, @telefono, @fechaNacimiento,
            @fotoIdCliente, @fotoPerfilCliente, @generoCliente, 1, 1);

    SET @ReturnValue = 0; -- Successfully inserted, set return value to 0
END;
