ALTER PROCEDURE [dbo].[sp_getMedicionByCliente]
    @idCliente VARCHAR(25)
AS
BEGIN
    SET NOCOUNT ON;
    
    SELECT *
    FROM TMedicion
    WHERE idCliente = @idCliente
	order by fecha desc;
END
