ALTER PROCEDURE [dbo].[sp_addEntrenamientoCompartido]
    @idEntrenamiento INT,
    @cedulaCliente NVARCHAR(25),
    @cedulaUsuario NVARCHAR(25),
    @ReturnValue INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    -- Verifica si ya existe un entrenamiento compartido con el mismo idEntrenamiento para cedulasUsuario
    IF EXISTS (SELECT 1 FROM TEntrenamientoCompartido WHERE idEntrenamiento = @idEntrenamiento AND cedulaUsuario = @cedulaUsuario)
    BEGIN
        SET @ReturnValue = -1; -- Ya existe un entrenamiento compartido con el mismo idEntrenamiento para esta cedulaUsuario
        RETURN; -- Salir del procedimiento
    END

    -- Inserta los datos en la tabla
    INSERT INTO TEntrenamientoCompartido (idEntrenamiento, cedulaCliente, cedulaUsuario)
    VALUES (@idEntrenamiento, @cedulaCliente, @cedulaUsuario);

    SET @ReturnValue = 0; -- Inserción exitosa, establecer valor de retorno a 0
END;

CREATE PROCEDURE [dbo].[sp_GetEntrenamientosCompartidosByCliente]
    @cedulaCliente NVARCHAR(25),
    @idEntrenamiento INT
AS
BEGIN
    SELECT 
        *
    FROM 
        TEntrenamientoCompartido
    WHERE 
        cedulaCliente = @cedulaCliente
        AND idEntrenamiento = @idEntrenamiento;
END

AS
BEGIN
